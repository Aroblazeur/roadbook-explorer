   237→    async function fetchJson(path, options = {}) {
   238→        const url = new URL(path, window.location.href);
   239→        if (options.forceReload) url.searchParams.set("t", String(Date.now()));
   240→        const response = await fetch(url.href, { cache: options.forceReload ? "reload" : "no-store" });
   241→        if (!response.ok) throw new Error(`HTTP ${response.status}`);
   242→        return response.json();
   243→    }

   245→    function normalizeRoadbookForEditing(roadbook, id) {
   246→        const clone = structuredClone(roadbook || {});
   247→        clone.id = safeText(clone.id, id);
   248→        clone.title = safeText(clone.title, id);
   249→        clone.description = safeText(clone.description, "");
   250→        clone.metadata = normalizeMetadataForEditing(clone.metadata);
   251→        clone.stages = extractEditableStages(clone);
   252→        clone.accommodation = normalizeRoadbookAccommodations(clone.accommodation);
   253→        clone.pois = normalizeRoadbookPois(clone.pois);
   254→        clone.notes = normalizeRoadbookNotes(clone.notes);
   255→        clone.contributions = normalizeContributions(clone.contributions);
   256→        delete clone.days;
   257→        return clone;
   258→    }

   272→    function extractEditableStages(roadbook) {

   282→    function normalizeStage(stage, index) {
   283→        const normalized = structuredClone(stage || {});
   284→        normalized.stage = toFiniteNumber(normalized.stage) ?? index + 1;
   285→        normalized.title = safeText(normalized.title, `Étape ${normalized.stage}`);
   286→        normalized.departure = safeText(normalized.departure, "");
   287→        normalized.arrival = safeText(normalized.arrival, "");
   288→        normalized.distance = toFiniteNumber(normalized.distance);
   289→        normalized.elevationGain = toFiniteNumber(normalized.elevationGain);
   290→        normalized.elevationLoss = toFiniteNumber(normalized.elevationLoss);
   291→        normalized.description = safeText(normalized.description, "");
   292→        STAGE_REFERENCE_FIELDS.forEach(field => {
   293→            normalized[field] = safeText(normalized[field], "");
   294→        });
   295→        normalized.warning = normalizeStringList(normalized.warning);
   296→        normalized.pois = normalizePois(normalized.pois || normalized.pointsOfInterest || normalized.interest);
   297→        normalized.pointsOfInterest = normalized.pois;
   298→        normalized.interest = normalized.pois;
   299→        normalized.accommodation = normalizeStageAccommodation(normalized.accommodation);
   300→        normalized.accommodationType = safeText(normalized.accommodationType, "");
   301→        normalized.noteItems = normalizeStageNoteItems(normalized.noteItems);
   302→        normalized.variants = normalizeVariants(
   303→            Array.isArray(normalized.variants) && normalized.variants.length
   304→                ? normalized.variants
   305→                : normalized.substeps
   306→        );
   307→        normalized.substeps = normalized.variants;
   308→        return normalized;
   309→    }

   447→    function normalizeContribution(contribution, index, usedIds = new Set()) {
   448→        const normalized = contribution && typeof contribution === "object" && !Array.isArray(contribution)
   449→            ? structuredClone(contribution)
   450→            : {};
   451→        normalized.id = reserveUniqueContributionId(
   452→            usedIds,
   453→            safeText(normalized.id, ""),
   454→            `contribution-${Date.now()}-${index + 1}`
   455→        );
   456→        normalized.type = safeText(normalized.type, "travelerNote");
   457→        normalized.stage = integerOrNull(normalized.stage);
   458→        normalized.createdAt = safeText(normalized.createdAt, "");
   459→        normalized.status = safeText(normalized.status, "");
   460→        normalized.source = safeText(normalized.source, "");
   461→        normalized.author = safeText(normalized.author, "");
   462→        normalized.timestamp = safeText(normalized.timestamp, "");
   463→        normalized.payload = normalizeContributionPayload(normalized.payload);
   464→        return normalized;
   465→    }

   467→    function reserveUniqueContributionId(usedIds, preferredId, fallbackPrefix = "contribution") {
   468→        const baseId = safeText(preferredId, "") || safeText(fallbackPrefix, "contribution");
   469→        if (!usedIds.has(baseId)) {
   470→            usedIds.add(baseId);
   471→            return baseId;
   472→        }
   473→        let suffix = 1;
   474→        let candidate = `${baseId}-${suffix}`;
   475→        while (usedIds.has(candidate)) {
   476→            suffix += 1;
   477→            candidate = `${baseId}-${suffix}`;
   478→        }
   479→        usedIds.add(candidate);
   480→        return candidate;
   481→    }

   693→        const keys = [];
   694→        const seen = new Set();
   695→
   696→        PRIMARY_METADATA_KEYS.forEach(key => {
   697→            keys.push(key);
   698→            seen.add(key);
   699→        });
   700→
   701→        Object.keys(metadata || {}).forEach(key => {
   702→            if (seen.has(key)) return;
   703→            if (!isEditableMetadataValue(metadata[key])) return;
   704→            keys.push(key);
   705→            seen.add(key);
   706→        });
   707→
   708→        return keys;
